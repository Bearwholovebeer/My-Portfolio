Script de Scan Réseau

Script Python qui scanne un réseau local, liste les hôtes actifs et détecte les ports ouverts les plus courants.

Python Scapy Socket 
#!/usr/bin/env python3
"""Scan réseau local - hôtes actifs + ports ouverts."""

import socket
import ipaddress
from scapy.all import ARP, Ether, srp

def discover_hosts(network: str) -> list[dict]:
    """ARP scan pour découvrir les hôtes actifs."""
    packet = Ether(dst="ff:ff:ff:ff:ff:ff") / ARP(pdst=network)
    result = srp(packet, timeout=2, verbose=0)[0]
    hosts = []
    for _, received in result:
        hosts.append({
            "ip": received.psrc,
            "mac": received.hwsrc
        })
    return hosts

def scan_ports(ip: str, ports: list[int]) -> list[int]:
    """Scan TCP connect sur une liste de ports."""
    open_ports = []
    for port in ports:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(0.5)
        if sock.connect_ex((ip, port)) == 0:
            open_ports.append(port)
        sock.close()
    return open_ports

COMMON_PORTS = [21, 22, 23, 53, 80, 443, 445, 3306, 3389, 8080]

if __name__ == "__main__":
    network = input("Réseau (ex: 192.168.1.0/24) : ")
    print(f"\n[*] Scan ARP sur {network}...")
    hosts = discover_hosts(network)
    print(f"[+] {len(hosts)} hôte(s) trouvé(s)\n")

    for host in hosts:
        ip = host["ip"]
        mac = host["mac"]
        open_p = scan_ports(ip, COMMON_PORTS)
        ports_str = ", ".join(map(str, open_p)) if open_p else "aucun"
        print(f"  {ip}  ({mac})  —  ports ouverts : {ports_str}")

Dashboard de Monitoring

Mini dashboard HTML/JS affichant en temps réel l'état fictif de services réseau : charge CPU, RAM et statut des services.

HTML CSS JavaScript 
// monitoring.js — Simulation de métriques réseau

const services = [
  { name: "Firewall",    status: "up" },
  { name: "DNS",         status: "up" },
  { name: "VPN Gateway", status: "up" },
  { name: "Web Server",  status: "up" },
  { name: "SIEM",        status: "up" },
];

function randomMetric(min, max) {
  return (Math.random() * (max - min) + min).toFixed(1);
}

function updateDashboard() {
  const cpu  = randomMetric(15, 85);
  const ram  = randomMetric(40, 78);
  const net  = randomMetric(10, 950);

  document.getElementById("cpu").textContent  = cpu + "%";
  document.getElementById("ram").textContent  = ram + "%";
  document.getElementById("net").textContent  = net + " Mbps";

  // Simuler un changement aléatoire de statut
  const idx = Math.floor(Math.random() * services.length);
  services[idx].status = Math.random() > 0.12 ? "up" : "down";

  const list = document.getElementById("services");
  list.innerHTML = services.map(s =>
    `<li class="${s.status}">${s.name} — ${s.status.toUpperCase()}</li>`
  ).join("");
}

setInterval(updateDashboard, 2000);
updateDashboard();
